\long\def\testglyph#1{%
\leavevmode\llap{#1:\glyphpath{#1}:}
\setbox\glyphbox\hbox{#1}%
\glyphtotalht=\dimexpr\the\ht\glyphbox+\the\dp\glyphbox\relax
% \showthe\glyphtotalht % --> works
\rule[-\the\dp\glyphbox]{\the\wd\glyphbox}{\dimexpr\the\ht\glyphbox+\the\dp\glyphbox}
\setbox\glyphtypesetbox=\vbox to \glyphtotalht{%
  % 1. use \rule as example:
  \hbox{\rule{\wd\glyphbox}{\the\glyphtotalht}}%
  % 2. test glyph "[x], [y], [Q], [']":
  \hbox{\glyphpath{#1}}
  \vss
}
\lower\the\dp\glyphbox\box\glyphtypesetbox\par
width: \the\wd\glyphbox;
height: \the\ht\glyphbox;
depth: \the\dp\glyphbox\par
}

\foreach \i in {a,b,...,z}
  {\testglyph\i}

\foreach \i in {A,B,...,Z}
  {\testglyph\i}
